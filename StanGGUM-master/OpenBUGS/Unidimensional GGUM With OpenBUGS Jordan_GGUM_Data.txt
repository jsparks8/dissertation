# How to run this model in OpenBUGS

# Step One:
# Model -> Specification
# Double click on "model"  in syntax to select that text-- ONLY select "model"
# Click "Check Model" and verify that the model is syntactically correct (check lower left hand corner)
# Double click on FIRST "list" in syntax to select -- ONLY select "list"
# Click "Load Data" and verify that the data is loaded (check lower left hand corner)
# Double click on the first "r" down where the responses are (so the r of r[,1]) to select it-- ONLY select that first "r"
# Click "Load Data" again and double check the data loaded
# Click "Compile" and wait until it's done -- You will be able to tell by, surprise, a message in the lower left hand corner
# Double click on SECOND "list" in syntax
# Click "Load Inits" and make sure it did that correctly

# Step Two: 
# Inference -> Samples
# In the white node text box, type the following, and after each one "set" the parameter:
# a
# b
# theta
# tau
# psi
# now type * in that box and just leave it there, don't hit set

# Step Three:
# Model -> Update
# Updates: 10,000
# Refresh: 1,000
# Make it do the thing and wait

# Lastly...
# So once it's done you can go back to the Sample Monitor Tool window and get statistics, 
# such as "statistics" "coda" etc etc to figure out how well your model did, 
# but you probably won't know what those things mean and that's OK


model GGUM                                                                                                          
{  
       
# specify prior distributions   
                                                                                                           
for (j in 1:J) {		                                                                                                      
   b[j,1] ~ dnorm(0, .25);                                                                                            
   a[j,1] ~ dnorm(0, .5)  I(.0,4.0);                                                                                       
                                                      

}                                                                                                                       

for (j in 1:J){                                                                                                         
	for (k in 2:K){                                                                                                        
           tau[j,k] ~ dlnorm(0, .04);                                                                         
}                                                                                                                       
}	

#specify model

for (j in 1:J){                                                                                                         
   tau[j,1] <- 0;                                                                                                       
   psi[j,1] <- 0;                                                                                                       
	for (k in 2:K){                                                                                                        
           psi[j,k] <- sum(a[j,1:D])*tau[j,k];                                                                          
}                                                                                                                       
}	                                                                                                                      
for (i in 1:N) {				                                                                                                    
   theta[i,1] ~ dnorm(0,1);                                                                                             
  for (j in 1:J) {	                                                                                                     
     for (z in 1:K) {                                                                                                   
        for (d in 1:D) {                                                                                                
            cpt1[i,j,z,d] <-  pow(a[j,d],2)*pow((theta[i,d]-b[j,d] ),2);                                                
            cpt2[i,j,z,d] <-  pow(a[j,d],2)*pow((theta[i,d]-b[j,d] ),2);                                                
        }                                                                                                               
        cp[i,j,z] <- exp((z-1)*sqrt(sum(cpt1[i,j,z,1:D])) + sum(psi[j,1:z])) +                                          
                       exp(((2*K)-z)*sqrt(sum(cpt2[i,j,z,1:D])) + sum(psi[j,1:z]));                                     
     }                                                                                                                  
     den[i,j] <- sum(cp[i,j,1:K]);                                                                                      
     for (z in 1:K){                                                                                                    
	prob[i,j,z] <- cp[i,j,z]/den[i,j];	                                                                                    
     }                                                                                                                  
     r[i,j] ~ dcat(prob[i,j,]);	                                                                                        
  }                                                                                                                     
}                                                                                                                       
}
